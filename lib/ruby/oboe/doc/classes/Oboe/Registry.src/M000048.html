    def Registry.has_scheme?(nid, scheme)

      # return true if nid == @@props['registry.nid._ri']
      has_scheme = false

      @@props['registry.nids'].scan(/\S+/) do |_nid|
        if nid == @@props[_nid]
            @@props["#{_nid}.schemes"].scan(/\S+/) do |_scheme|
            has_scheme = true if scheme == @@props[_scheme]
          end
        end
      end

      return has_scheme
    end
